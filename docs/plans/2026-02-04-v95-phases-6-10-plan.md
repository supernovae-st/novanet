### 10.2 Terminal color mode detection

```rust
fn detect_color_mode() -> ColorMode {
    match std::env::var("COLORTERM").as_deref() {
        Ok("truecolor") | Ok("24bit") => ColorMode::TrueColor,
        _ => match std::env::var("TERM").as_deref() {
            Ok(t) if t.contains("256color") => ColorMode::Color256,
            _ => ColorMode::Color16,
        }
    }
}
```

### 10.3 Use palette_256/palette_16

```rust
fn get_realm_color(realm: &str, mode: ColorMode) -> Color {
    match mode {
        ColorMode::TrueColor => Color::from_hex(&TAXONOMY.realms[realm].color),
        ColorMode::Color256 => Color::Indexed(PALETTE_256.realm(realm)),
        ColorMode::Color16 => Color::from_ansi(PALETTE_16.realm(realm)),
    }
}
```

### 10.4 Use trait_borders.unicode_char

```rust
fn draw_node_border(trait_key: &str) -> char {
    TRAIT_BORDERS.iter()
        .find(|(k, _)| *k == trait_key)
        .map(|(_, c)| *c)
        .unwrap_or('─')
}
```
