**Step 1: Create org-config.yaml**

```yaml
# packages/core/models/node-classes/org/config/org-config.yaml
# OrgConfig - Organization configuration root node
# v11.3: Merged from Organization + Tenant

node:
  name: OrgConfig
  realm: org
  layer: config
  trait: invariant
  icon: "🏢"
  description: "Organization configuration root. Parent of all projects."

  standard_properties:
    key:
      type: string
      required: true
      description: "Unique organization identifier (kebab-case)"
      example: "supernovae"

    display_name:
      type: string
      required: true
      description: "Human-readable organization name"
      example: "SuperNovae"

    description:
      type: string
      required: true
      description: "Brief description of the organization"
      example: "AI-powered content localization"

    llm_context:
      type: string
      required: true
      description: "LLM routing context (USE/TRIGGERS/NOT format)"
      example: "USE: org-level context. TRIGGERS: supernovae, company. NOT: project-specific."

    created_at:
      type: datetime
      required: true

    updated_at:
      type: datetime
      required: true

  properties:
    domain:
      type: string
      required: false
      description: "Primary domain for the organization"
      example: "supernovae.studio"

  relations:
    HAS_PROJECT:
      to: Project
      cardinality: "1:N"
      description: "Projects owned by this organization"

  example:
    data:
      key: "supernovae"
      display_name: "SuperNovae"
      description: "AI-powered content localization"
      llm_context: "USE: org-level context. TRIGGERS: supernovae, company. NOT: project-specific."
      domain: "supernovae.studio"

    cypher: |
      MATCH (org:OrgConfig {key: $key})
      OPTIONAL MATCH (org)-[:HAS_PROJECT]->(p:Project)
      RETURN org, collect(DISTINCT p) AS projects
```
